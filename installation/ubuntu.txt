
sudo apt-get update
sudo apt install wget make gcc tcl build-essential
JAVA 11 Only - sudo apt-get install openjdk-11-jdk
Go to root directory: cd /
Make Cassandra Workstation: mkdir apache_cassandra
Download Cassandra Binaries:  wget https://dlcdn.apache.org/cassandra/4.1.4/apache-cassandra-4.1.4-bin.tar.gz
Extract Binaries: tar -xvf apache-cassandra-4.1.4-bin.tar.gz
Go into extracted directory: cd apache-cassandra-4.1.4/
Start Cassandra: bin/cassandra -f -R
